int sum(int x,int y){  //sum
    return x+y;
}
int difference(int x,int y){
    return x-y;//minus
}
int product(int x,int y){   //multi
    return (x*y);
}
int div(int x,int y)
{
    return x/y; //div
}
int main(){
    int a,b,c,d,e,result=0;
    a=getint();
    b=getint();
    c=getint();
    d=getint();
    e=getint();
    e=-e+c*d-a%b+c/(a+b)+sum(a,b)-difference(c,d)*product(a,b)+div(a,c);
    result=result+e*a;
    printf("result is %d.",result);
    return 0;
}
